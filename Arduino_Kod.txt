// SmartSafe Guardian: RFID + serwo/zamek + czerwona LED + buzzer aktywny + alarm + Processing

#include <SPI.h>
#include <MFRC522.h>
#include <Servo.h>

// ===================== PINY =====================

#define RST_PIN 9
#define SS_PIN 10
#define SERVO_PIN 3

#define RED_LED_PIN 4
#define BUZZER_PIN 6

// ===================== OBIEKTY =====================

MFRC522 mfrc522(SS_PIN, RST_PIN);
Servo doorServo;

// ===================== KONFIGURACJA ZAMKA =====================

const int LOCKED_ANGLE = 0;
const int UNLOCKED_ANGLE = 180;

const unsigned long UNLOCK_DURATION_MS = 5000;
const unsigned long RFID_COOLDOWN_MS = 800;

// ===================== KONFIGURACJA SYGNAŁÓW =====================

const unsigned long GRANTED_BEEP_MS = 120;
const unsigned long DENIED_BEEP_MS = 250;
const unsigned long DENIED_LED_MS = 1000;
const unsigned long ALARM_PULSE_MS = 300;
const unsigned long DENIED_STATE_MS = 1000;

// ===================== AUTORYZOWANE UID =====================
// TUTAJ WPISZ PRAWDZIWE UID SWOJEJ KARTY I BRELOKA.
// UID wpisuj DUŻYMI LITERAMI, bez spacji, np. "A1B2C3D4".

const char* AUTHORIZED_UIDS[] = {
  "43DB6419",   // karta - zmień na swój UID, jeśli jest inny
  "11223344"    // brelok - zmień na swój UID
};

const byte AUTHORIZED_UIDS_COUNT = sizeof(AUTHORIZED_UIDS) / sizeof(AUTHORIZED_UIDS[0]);

// ===================== DODAWANE KARTY W RAM =====================
// Karty dodane przyciskiem ADD LAST CARD znikają po restarcie Arduino.

const byte MAX_DYNAMIC_CARDS = 10;
const byte MAX_UID_LENGTH = 20;

char dynamicUIDs[MAX_DYNAMIC_CARDS][MAX_UID_LENGTH + 1];
byte dynamicCardsCount = 0;

// ===================== STANY SYSTEMU =====================

enum SafeState {
  STATE_LOCKED,
  STATE_UNLOCKED,
  STATE_ACCESS_GRANTED,
  STATE_ACCESS_DENIED,
  STATE_ALARM
};

SafeState currentState = STATE_LOCKED;

// ===================== ZMIENNE SYSTEMOWE =====================

const byte MAX_FAILED_ATTEMPTS = 3;

byte failedAttempts = 0;
bool alarmActive = false;
bool doorLocked = true;
bool autoLockEnabled = false;

String lastUID = "-";
String lastAuthResult = "-";

unsigned long unlockUntil = 0;
unsigned long lastRFIDReadMs = 0;
unsigned long returnToLockedAt = 0;

unsigned long buzzerUntil = 0;
unsigned long redLedUntil = 0;
unsigned long alarmPulseLastToggle = 0;
bool alarmPulseOutputOn = false;

// ===================== SERIAL =====================

String serialBuffer = "";

// ===================== SETUP =====================

void setup() {
  pinMode(RED_LED_PIN, OUTPUT);
  pinMode(BUZZER_PIN, OUTPUT);

  stopPhysicalSignals();

  Serial.begin(9600);

  SPI.begin();
  mfrc522.PCD_Init();

  doorServo.attach(SERVO_PIN);
  lockSafe(false);

  sendMessage("LOG", "System started");
  sendMessage("LOG", "Dynamic cards are stored in RAM");
  sendFullStatus();
}

// ===================== LOOP =====================

void loop() {
  readSerialCommands();
  updateAutoLock();
  updateTemporaryState();
  updatePhysicalSignals();
  checkRFID();
}

// ===================== RFID =====================

void checkRFID() {
  if (!mfrc522.PICC_IsNewCardPresent()) {
    return;
  }

  if (!mfrc522.PICC_ReadCardSerial()) {
    return;
  }

  if (millis() - lastRFIDReadMs < RFID_COOLDOWN_MS) {
    haltRFID();
    return;
  }

  lastRFIDReadMs = millis();

  String uid = readUID();
  lastUID = uid;

  sendMessage("LOG", "RFID UID scanned: " + uid);

  if (isAuthorizedUID(uid)) {
    handleAccessGranted(uid);
  } else {
    handleAccessDenied(uid);
  }

  haltRFID();
}

String readUID() {
  String uid = "";

  for (byte i = 0; i < mfrc522.uid.size; i++) {
    if (mfrc522.uid.uidByte[i] < 0x10) {
      uid += "0";
    }

    uid += String(mfrc522.uid.uidByte[i], HEX);
  }

  uid.toUpperCase();
  return uid;
}

bool isAuthorizedUID(const String& uid) {
  for (byte i = 0; i < AUTHORIZED_UIDS_COUNT; i++) {
    if (uid == AUTHORIZED_UIDS[i]) {
      return true;
    }
  }

  for (byte i = 0; i < dynamicCardsCount; i++) {
    if (uid == dynamicUIDs[i]) {
      return true;
    }
  }

  return false;
}

void haltRFID() {
  mfrc522.PICC_HaltA();
  mfrc522.PCD_StopCrypto1();
}

// ===================== OBSŁUGA DOSTĘPU =====================

void handleAccessGranted(const String& uid) {
  if (alarmActive) {
    clearAlarmByAuthorizedCard();
  }

  failedAttempts = 0;
  lastAuthResult = "GRANTED";
  returnToLockedAt = 0;

  startGrantedSignal();

  setState(STATE_ACCESS_GRANTED);
  sendRFIDEvent(uid, "GRANTED");
  sendMessage("ATTEMPTS", String(failedAttempts));
  sendMessage("ALARM", "OFF");
  sendMessage("LOG", "Access granted");

  unlockSafe(true);
}

void handleAccessDenied(const String& uid) {
  lastAuthResult = "DENIED";

  sendRFIDEvent(uid, "DENIED");

  if (alarmActive) {
    sendMessage("ATTEMPTS", String(failedAttempts));
    sendMessage("ALARM", "ACTIVE");
    sendMessage("LOG", "Access denied - alarm active");
    return;
  }

  failedAttempts++;

  startDeniedSignal();

  setState(STATE_ACCESS_DENIED);
  returnToLockedAt = millis() + DENIED_STATE_MS;

  sendMessage("ATTEMPTS", String(failedAttempts));
  sendMessage("LOG", "Access denied");

  if (failedAttempts >= MAX_FAILED_ATTEMPTS) {
    triggerAlarm("WRONG_CARD_LIMIT");
  }
}

// ===================== SERWO / ZAMEK =====================

void unlockSafe(bool enableAutoLock) {
  if (alarmActive) {
    sendMessage("LOG", "Unlock rejected - alarm active");
    sendMessage("ALARM", "ACTIVE");
    return;
  }

  doorServo.write(UNLOCKED_ANGLE);

  doorLocked = false;
  autoLockEnabled = enableAutoLock;

  if (enableAutoLock) {
    unlockUntil = millis() + UNLOCK_DURATION_MS;
  }

  setState(STATE_UNLOCKED);
  sendMessage("DOOR", "UNLOCKED");
  sendMessage("LOG", "Door unlocked");
}

void lockSafe(bool sendLog) {
  doorServo.write(LOCKED_ANGLE);

  doorLocked = true;
  autoLockEnabled = false;
  returnToLockedAt = 0;

  sendMessage("DOOR", "LOCKED");

  if (alarmActive) {
    setState(STATE_ALARM);
  } else {
    setState(STATE_LOCKED);
  }

  if (sendLog) {
    sendMessage("LOG", "Door locked");
  }
}

void updateAutoLock() {
  if (!autoLockEnabled) {
    return;
  }

  if (millis() >= unlockUntil) {
    lockSafe(true);
  }
}

void updateTemporaryState() {
  if (returnToLockedAt == 0) {
    return;
  }

  if (alarmActive) {
    returnToLockedAt = 0;
    return;
  }

  if (millis() >= returnToLockedAt && doorLocked) {
    returnToLockedAt = 0;
    setState(STATE_LOCKED);
  }
}

// ===================== ALARM I SYGNAŁY FIZYCZNE =====================

void triggerAlarm(const char* reason) {
  alarmActive = true;
  failedAttempts = MAX_FAILED_ATTEMPTS;
  autoLockEnabled = false;
  returnToLockedAt = 0;

  doorServo.write(LOCKED_ANGLE);
  doorLocked = true;

  alarmPulseOutputOn = true;
  alarmPulseLastToggle = millis();
  digitalWrite(RED_LED_PIN, HIGH);
  digitalWrite(BUZZER_PIN, HIGH);

  sendMessage("DOOR", "LOCKED");
  setState(STATE_ALARM);
  sendMessage("ATTEMPTS", String(failedAttempts));
  sendMessage("ALARM", reason);

  String logMessage = "Alarm triggered: ";
  logMessage += reason;
  sendMessage("LOG", logMessage);
}

void resetAlarm() {
  alarmActive = false;
  failedAttempts = 0;
  lastAuthResult = "-";
  returnToLockedAt = 0;

  stopPhysicalSignals();
  lockSafe(false);

  sendMessage("ATTEMPTS", String(failedAttempts));
  sendMessage("ALARM", "RESET");
  sendMessage("LOG", "Alarm reset");
  sendFullStatus();
}

void clearAlarmByAuthorizedCard() {
  alarmActive = false;
  failedAttempts = 0;
  returnToLockedAt = 0;

  stopPhysicalSignals();

  sendMessage("ATTEMPTS", String(failedAttempts));
  sendMessage("ALARM", "RESET");
  sendMessage("LOG", "Alarm reset by authorized card");
}

void startGrantedSignal() {
  buzzerUntil = millis() + GRANTED_BEEP_MS;
}

void startDeniedSignal() {
  redLedUntil = millis() + DENIED_LED_MS;
  buzzerUntil = millis() + DENIED_BEEP_MS;
}

void updatePhysicalSignals() {
  unsigned long now = millis();

  if (alarmActive) {
    if (now - alarmPulseLastToggle >= ALARM_PULSE_MS) {
      alarmPulseLastToggle = now;
      alarmPulseOutputOn = !alarmPulseOutputOn;

      digitalWrite(RED_LED_PIN, alarmPulseOutputOn ? HIGH : LOW);
      digitalWrite(BUZZER_PIN, alarmPulseOutputOn ? HIGH : LOW);
    }

    return;
  }

  digitalWrite(RED_LED_PIN, now < redLedUntil ? HIGH : LOW);
  digitalWrite(BUZZER_PIN, now < buzzerUntil ? HIGH : LOW);
}

void stopPhysicalSignals() {
  buzzerUntil = 0;
  redLedUntil = 0;
  alarmPulseOutputOn = false;

  digitalWrite(RED_LED_PIN, LOW);
  digitalWrite(BUZZER_PIN, LOW);
}

// ===================== DODAWANIE KART W RAM =====================

void addLastUIDToRAM() {
  if (lastUID == "-" || lastUID.length() == 0) {
    sendMessage("CARD_ADD", "NO_LAST_UID");
    sendMessage("LOG", "No UID to add");
    return;
  }

  if (!isValidUIDFormat(lastUID)) {
    Serial.print("CARD_ADD|INVALID_UID|");
    Serial.println(lastUID);
    sendMessage("LOG", "Invalid UID format");
    return;
  }

  if (isAuthorizedUID(lastUID)) {
    Serial.print("CARD_ADD|ALREADY_EXISTS|");
    Serial.println(lastUID);
    sendMessage("CARDS_COUNT", String(dynamicCardsCount));
    sendMessage("LOG", "UID already authorized");
    return;
  }

  if (dynamicCardsCount >= MAX_DYNAMIC_CARDS) {
    sendMessage("CARD_ADD", "MEMORY_FULL");
    sendMessage("LOG", "Dynamic card memory full");
    return;
  }

  lastUID.toCharArray(dynamicUIDs[dynamicCardsCount], MAX_UID_LENGTH + 1);
  dynamicCardsCount++;

  Serial.print("CARD_ADD|OK|");
  Serial.println(lastUID);

  sendMessage("CARDS_COUNT", String(dynamicCardsCount));
  sendMessage("LOG", "UID added to RAM");
}

void listCardsFromRAM() {
  sendMessage("CARDS_COUNT", String(dynamicCardsCount));

  if (dynamicCardsCount == 0) {
    sendMessage("LOG", "No dynamic cards in RAM");
    return;
  }

  for (byte i = 0; i < dynamicCardsCount; i++) {
    Serial.print("CARD|");
    Serial.print(i + 1);
    Serial.print("|");
    Serial.println(dynamicUIDs[i]);
  }

  sendMessage("LOG", "Dynamic card list sent");
}

void clearCardsFromRAM() {
  for (byte i = 0; i < MAX_DYNAMIC_CARDS; i++) {
    dynamicUIDs[i][0] = '\0';
  }

  dynamicCardsCount = 0;

  sendMessage("CARDS_CLEAR", "OK");
  sendMessage("CARDS_COUNT", String(dynamicCardsCount));
  sendMessage("LOG", "Dynamic cards cleared from RAM");
}

bool isValidUIDFormat(const String& uid) {
  int len = uid.length();

  if (len < 4 || len > MAX_UID_LENGTH) {
    return false;
  }

  for (int i = 0; i < len; i++) {
    char c = uid.charAt(i);

    bool isDigit = c >= '0' && c <= '9';
    bool isUpperHex = c >= 'A' && c <= 'F';
    bool isLowerHex = c >= 'a' && c <= 'f';

    if (!isDigit && !isUpperHex && !isLowerHex) {
      return false;
    }
  }

  return true;
}

// ===================== KOMUNIKACJA: PROCESSING -> ARDUINO =====================

void readSerialCommands() {
  while (Serial.available() > 0) {
    char incomingChar = Serial.read();

    if (incomingChar == '\r') {
      continue;
    }

    if (incomingChar == '\n') {
      serialBuffer.trim();

      if (serialBuffer.length() > 0) {
        processCommand(serialBuffer);
      }

      serialBuffer = "";
    } else {
      serialBuffer += incomingChar;

      if (serialBuffer.length() > 100) {
        serialBuffer = "";
        sendMessage("LOG", "Serial buffer cleared");
      }
    }
  }
}

void processCommand(String command) {
  command.trim();

  sendMessage("LOG", "Command received: " + command);

  if (command == "CMD|LOCK") {
    lockSafe(true);
  }
  else if (command == "CMD|UNLOCK") {
    unlockSafe(true);
  }
  else if (command == "CMD|RESET_ALARM") {
    resetAlarm();
  }
  else if (command == "CMD|STATUS") {
    sendFullStatus();
  }
  else if (command == "CMD|ADD_LAST_UID") {
    addLastUIDToRAM();
  }
  else if (command == "CMD|LIST_CARDS") {
    listCardsFromRAM();
  }
  else if (command == "CMD|CLEAR_CARDS") {
    clearCardsFromRAM();
  }
  else {
    sendMessage("LOG", "Unknown command: " + command);
  }
}

// ===================== KOMUNIKACJA: ARDUINO -> PROCESSING =====================

void sendMessage(const String& type, const String& value) {
  Serial.print(type);
  Serial.print("|");
  Serial.println(value);
}

void sendRFIDEvent(const String& uid, const String& result) {
  Serial.print("RFID|");
  Serial.print(uid);
  Serial.print("|");
  Serial.println(result);
}

void setState(SafeState newState) {
  currentState = newState;
  sendMessage("STATE", stateToString(currentState));
}

String stateToString(SafeState state) {
  switch (state) {
    case STATE_LOCKED:
      return "LOCKED";

    case STATE_UNLOCKED:
      return "UNLOCKED";

    case STATE_ACCESS_GRANTED:
      return "ACCESS_GRANTED";

    case STATE_ACCESS_DENIED:
      return "ACCESS_DENIED";

    case STATE_ALARM:
      return "ALARM";

    default:
      return "UNKNOWN";
  }
}

void sendFullStatus() {
  sendMessage("STATE", stateToString(currentState));

  if (doorLocked) {
    sendMessage("DOOR", "LOCKED");
  } else {
    sendMessage("DOOR", "UNLOCKED");
  }

  sendMessage("ATTEMPTS", String(failedAttempts));

  if (alarmActive) {
    sendMessage("ALARM", "ACTIVE");
  } else {
    sendMessage("ALARM", "OFF");
  }

  sendMessage("CARDS_COUNT", String(dynamicCardsCount));

  if (lastUID != "-") {
    sendRFIDEvent(lastUID, lastAuthResult);
  }

  sendMessage("LOG", "Status sent");
}
